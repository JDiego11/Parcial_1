#include <iostream>

using namespace std;

int main()
{
    //ESTO ES UNA IDEA INICIAL PARA LA FUNCION IMAGEN//

    bool Matriz[8][8]; //Tama√±o para la matriz estandar
    int Condition;
    cout <<"Recuerde que para ingresar el patron:" << endl << ".1 reperesenta encendido." << endl << ".0 representa apagado" << endl;
    for (int j = 0; j < 8; j++){
        for (int i = 0; i < 8; i++){
            cout <<"Ingrese la condicion para el led " <<j+1<<", "<<i+1<<": ";
            cin >> Condition;
            if (Condition == 1){
                Matriz[j][i] = true;

    }
    for (int k = 0; k < 8; k++){
        for (int w = 0; w < 8; w++){
            cout << Matriz[k][w] << " ";
        }

    }
